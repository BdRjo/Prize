   638→   638→function GenericCRUDView<T extends { id: string }>({ 
   639→   639→  collection, title, columns, renderForm, newItemTemplate 
   640→   640→}: { 
   641→   641→  collection: string; title: string; columns: { key: string; label: string; render?: (item: T) => React.ReactNode }[];
   642→   642→  renderForm: (item: T, onChange: (item: T) => void) => React.ReactNode;
   643→   643→  newItemTemplate: T
   644→   644→}) {
   645→   645→  const [items, setItems] = useState<T[]>([])
   646→   646→  const [editing, setEditing] = useState<T | null>(null)
   647→   647→  const [isAdding, setIsAdding] = useState(false)
   648→   648→  const [form, setForm] = useState<T>(newItemTemplate)
   649→   649→  const [loading, setLoading] = useState(true)
   650→   650→
   651→   651→  const load = useCallback(async () => {
   652→   652→    try {
   653→   653→      const data = await apiFetch(collection)
   654→   654→      setItems(data)
   655→   655→    } catch {}
   656→   656→    setLoading(false)
   657→   657→  }, [collection])
   658→   658→
   659→   659→  useEffect(() => { load() }, [load])
   660→   660→
   661→   661→  const handleSave = async () => {
   662→   662→    try {
   663→   663→      if (isAdding) {
   664→   664→        await apiFetch(collection, { method: 'POST', body: JSON.stringify(form) })
   665→   665→      } else if (editing) {
   666→   666→        await apiFetch(collection, { method: 'PUT', body: JSON.stringify(form) })
   667→   667→      }
   668→   668→      setEditing(null)
   669→   669→      setIsAdding(false)
   670→   670→      setForm(newItemTemplate)
   671→   671→      load()
   672→   672→    } catch {}
   673→   673→  }
   674→   674→
   675→   675→  const handleDelete = async (id: string) => {
   676→   676→    if (!confirm('هل أنت متأكد من الحذف؟')) return
   677→   677→    try {
   678→   678→      await apiFetch(`${collection}?id=${id}`, { method: 'DELETE' })
   679→   679→      load()
   680→   680→    } catch {}
   681→   681→  }
   682→   682→
   683→   683→  const handleEdit = (item: T) => {
   684→   684→    setEditing(item)
   685→   685→    setForm(item)
   686→   686→    setIsAdding(false)
   687→   687→  }
   688→   688→
   689→   689→  const handleAdd = () => {
   690→   690→    setEditing(newItemTemplate)
   691→   691→    setForm({ ...newItemTemplate })
   692→   692→    setIsAdding(true)
   693→   693→  }
   694→   694→
   695→   695→  return (
   696→   696→    <div>
   697→   697→      <h2 className="text-2xl font-bold mb-6">{title}</h2>
   698→   698→      <DataTable items={items} columns={columns} onEdit={handleEdit} onDelete={handleDelete} onAdd={handleAdd} addLabel={`إضافة ${title}`} />
   699→   699→      <FormModal title={isAdding ? `إضافة ${title}` : `تعديل ${title}`} isOpen={!!editing} onClose={() => { setEditing(null); setIsAdding(false) }}>
   700→   700→        {renderForm(form, setForm)}
   701→   701→        <div className="flex gap-3 mt-6">
   702→   702→          <button onClick={handleSave} className="flex items-center gap-2 px-5 py-2.5 bg-[#1B5E20] text-white rounded-lg hover:bg-[#2E7D32]">
   703→   703→            <Icons name="save" className="w-4 h-4" /> حفظ
   704→   704→          </button>
   705→   705→          <button onClick={() => { setEditing(null); setIsAdding(false) }} className="px-5 py-2.5 bg-gray-200 rounded-lg hover:bg-gray-300">إلغاء</button>
   706→   706→        </div>
   707→   707→      </FormModal>
   708→   708→    </div>
   709→   709→  )
   710→   710→}

   715→   715→function BuilderView() {

   717→   717→  const [currentPage, setCurrentPage] = useState<PageItem | null>(null)

   730→   730→    const newBlock: BuilderBlock = {
   731→   731→      id: `block-${Date.now()}`,
   732→   732→      type,
   733→   733→      content: type === 'heading' ? 'عنوان جديد' : type === 'paragraph' ? 'نص الفقرة هنا...' : '',
   734→   734→      properties: {},

   740→   740→  const updateBlock = (id: string, updates: Partial<BuilderBlock>) => {
   741→   741→    setBlocks(blocks.map(b => b.id === id ? { ...b, ...updates } : b))
   742→   742→  }
   743→   743→
   744→   744→  const removeBlock = (id: string) => {
   745→   745→    setBlocks(blocks.filter(b => b.id !== id))

   749→   749→    const index = blocks.findIndex(b => b.id === id)
   750→   750→    if ((direction === 'up' && index === 0) || (direction === 'down' && index === blocks.length - 1)) return
   751→   751→    const newBlocks = [...blocks]
   752→   752→    const swapIndex = direction === 'up' ? index - 1 : index + 1
   753→   753→    ;[newBlocks[index], newBlocks[swapIndex]] = [newBlocks[swapIndex], newBlocks[index]]

   822→   822→      </div>
   823→   823→
